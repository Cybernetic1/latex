\begin{frame}
	\frametitle{Freedom of speech}
	Regarding posts and comments:
	\begin{enumerate}
		\item All posts / comments are recorded, publicly viewable and searchable
		\item Comments can be hidden but with open explanations
		\item Hidden comments can be appealed
		\item Ranking of comments by transparent democratic voting
	\end{enumerate}
\end{frame}

\begin{frame}
	\frametitle{Comments voting, ranking and sorting}
	\begin{itemize}
		\item Commenters and their comments cab be sorted by clustering algorithm
		\item Votes may be of type "like", "dislike", "inappropriate", etc.
		\item Commenters and comments can be ranked by weighted voting.
		\item Rewards can be distributed to viewers accordingly.
	\end{itemize}
\end{frame}
